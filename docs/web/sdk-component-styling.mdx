Geocortex Web Components support fully custom styling using CSS. However, by using Geocortex Web UI Elements, you can take advantage of styled, well tested, UI components that already participate in the [application theme](sdk-theme.mdx).

:::tip
If manual CSS styling is required, you can still participate in the application theme by [using the theme colours defined in the `BrandingService`](sdk-theme.mdx#using-theme-colors-in-custom-components).
:::

```tsx
import React from "react";
import {
    LayoutElement,
    LayoutElementProperties,
} from "@geocortex/web/components";
import List from "@geocortex/web/ui/list";
import ListItem from "@geocortex/web/ui/list-item";
import TitleBar from "@geocortex/web/ui/title-bar";
import Button from "@geocortex/web/ui/button";
import classNames from "classnames";
import ExampleComponentModel from "./ExampleComponentModel";
import "./ExampleComponent.css";

export default function ExampleComponent(
    props: LayoutElementProperties<ExampleComponentModel>
) {
    return (
        <LayoutElement {...props}>
            <List dense={true}>
                <TitleBar>Some Title</TitleBar>
                <ListItem>
                    <Button>Item 1</Button>
                </ListItem>
                <ListItem>
                    <Button>Item 2</Button>
                </ListItem>
                <ListItem>
                    <Button>Item 2</Button>
                </ListItem>
                <div className={classNames("red")}>
                    {" "}
                    I have a red background
                </div>
            </List>
        </LayoutElement>
    );
}
```

```tsx title="src/components/
import React from "react";
import { LayoutElement } from "@geocortex/web/components";
import DynamicIcon from "@geocortex/web/ui/icon/DynamicIcon";
import IconButton from "@geocortex/web/ui/icon-button";
import MenuItem from "@geocortex/web/ui/menu-item";
import MenuItemSecondaryAction from "@geocortex/web/ui/menu-list-item-secondary-action";
import ListItemText from "@geocortex/web/ui/list-item-text";

export default function ComponentWithIcon(props) {
    return (
        <LayoutElement {...props} style={{ backgroundColor: "white" }}>
            <MenuItem>
                <ListItemText
                    primary={"Icon Example"}
                    secondary={"cloud-download icon"}
                />
                <MenuItemSecondaryAction>
                    // highlight-start
                    <IconButton title="IconExample">
                        <DynamicIcon src="cloud-download" size="large" />
                    </IconButton>
                    // highlight-end
                </MenuItemSecondaryAction>
            </MenuItem>
        </LayoutElement>
    );
}
```
